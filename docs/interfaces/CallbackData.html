    Interface CallbackData

    Detached callback payload; optional fields are channel-dependent. Native MIP node and LP-iteration counts are int64 values and therefore use bigint.

    interface CallbackData {
        cut_pool?: {
            indices: Int32Array;
            lower: Float64Array;
            numCols: number;
            numCuts: number;
            starts: Int32Array;
            upper: Float64Array;
            values: Float64Array;
        };
        ipm_iteration_count?: number;
        log_type?: number;
        mip_dual_bound?: number;
        mip_gap?: number;
        mip_node_count?: bigint;
        mip_primal_bound?: number;
        mip_solution?: Float64Array<ArrayBufferLike>;
        mip_total_lp_iterations?: bigint;
        objective_function_value?: number;
        running_time?: number;
        simplex_iteration_count?: number;
        readonly [key: string]: unknown;
    }
